        document.getElementById('registrationForm').addEventListener('submit', function(e) {

            const formData = {

                name: document.getElementById('name').value,

                email: document.getElementById('email').value,

                phone: document.getElementById('phone').value,

                age: document.getElementById('age').value,

                gender: document.getElementById('gender').value,

                emergencyContact: document.getElementById('emergencyContact').value,

                registrationDate: new Date().toLocaleString(),

                event: 'Rodada de la Independencia MTB',

                distance: '15km',

                date: '31 de Agosto 2025',

                fee: '$350 MXN'

            };

           

            // Guardar en Excel

            saveToExcel(formData);

           

            // Mostrar confirmación

            document.getElementById('confirmation').style.display = 'block';

           

            // Desplazarse a la confirmación

            document.getElementById('confirmation').scrollIntoView({ behavior: 'smooth' });

           

            // Reiniciar formulario

            document.getElementById('registrationForm').reset();

        });

       

        function saveToExcel(data) {

            // Crear un libro de trabajo

            const wb = XLSX.utils.book_new();

           

            // Intentar cargar datos existentes

            let existingData = [];

            const storedData = localStorage.getItem('mtbRegistrations');

           

            if (storedData) {

                existingData = JSON.parse(storedData);

            }

           

            // Agregar nuevos datos

            existingData.push(data);

           

            // Guardar en localStorage

            localStorage.setItem('mtbRegistrations', JSON.stringify(existingData));

           

            // Convertir a hoja de cálculo

            const ws = XLSX.utils.json_to_sheet(existingData);

           

            // Agregar hoja al libro

            XLSX.utils.book_append_sheet(wb, ws, "Registros MTB");

           

            // Guardar archivo Excel

            XLSX.writeFile(wb, "registros_rodada_independencia.xlsx");

        }
